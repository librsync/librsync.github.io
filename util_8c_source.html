79 int rs_long_sqrt(rs_long_t v)
80 {
81  rs_long_t n, b;
82 
83  /* Find the most significant bit of the root. */
84  for (b = 1, n = v; (n >>= 2); b <<= 1) ;
85  /* Walk down the bits of the root. */
86  for (n = 0; b; b >>= 1) {
87  /* Set the bit in the answer n. */
88  n |= b;
89  /* If n^2 is too big, clear the bit. */
90  if (n * n > v)
91  n ^= b;
92  }
93  return (int)n;
94 }
