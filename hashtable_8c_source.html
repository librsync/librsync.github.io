26 /* Open addressing works best if it can take advantage of memory caches using
27  * locality for probes of adjacent buckets on collisions. So we pack the keys
28  * tightly together in their own key table and avoid referencing the element
29  * table and elements as much as possible. Key value zero is reserved as a
30  * marker for an empty bucket to avoid checking for NULL in the element table.
31  * If we do get a hash value of zero, we -1 to wrap it around to 0xffff. */
32 
33 /* Use max 0.8 load factor to avoid bad open addressing performance. */
34 #define HASHTABLE_LOADFACTOR_NUM 8
35 #define HASHTABLE_LOADFACTOR_DEN 10
36 
37 hashtable_t *hashtable_new(int size, hash_f hash, cmp_f cmp)
38 {
39  hashtable_t *t;
40  int size2;
41 
42  /* Adjust requested size to account for max load factor. */
43  size = 1 + size * HASHTABLE_LOADFACTOR_DEN / HASHTABLE_LOADFACTOR_NUM;
44  /* Use next power of 2 larger than the requested size. */
45  for (size2 = 1; size2 < size; size2 <<= 1) ;
46  if (!(t = calloc(1, sizeof(hashtable_t)+ size2 * sizeof(unsigned))))
47  return NULL;
48  if (!(t->etable = calloc(size2, sizeof(void *)))) {
49  free(t);
50  return NULL;
51  }
52  t->size = size2;
53  t->count = 0;
54  t->hash = hash;
55  t->cmp = cmp;
56 #ifndef HASHTABLE_NSTATS
57  t->find_count = t->match_count = t->hashcmp_count = t->entrycmp_count = 0;
58 #endif
59  return t;
60 }

70 /* MurmurHash3 finalization mix function. */
71 static inline unsigned mix32(unsigned int h)
72 {
73  h ^= h >> 16;
74  h *= 0x85ebca6b;
75  h ^= h >> 13;
76  h *= 0xc2b2ae35;
77  h ^= h >> 16;
78  return h;
79 }

100 void *hashtable_add(hashtable_t *t, void *e)
101 {
102  assert(e != NULL);
103  do_probe(t, e, k) {
104  if (!t->ktable[i]) {
105  t->count++;
106  t->ktable[i] = k;
107  return t->etable[i] = e;
108  }
109  } while_probe;
110  return NULL;
111 }

120 void *hashtable_find(hashtable_t *t, void *m)
121 {
122  assert(m != NULL);
123  void *e;
124  unsigned ke;
125 
126  stats_inc(t->find_count);
127  do_probe(t, m, km) {
128  if (!(ke = t->ktable[i]))
129  return NULL;
130  stats_inc(t->hashcmp_count);
131  if (km == ke) {
132  stats_inc(t->entrycmp_count);
133  if (!t->cmp(m, e = t->etable[i])) {
134  stats_inc(t->match_count);
135  return e;
136  }
137  }
138  } while_probe;
139  return NULL;
140 }
