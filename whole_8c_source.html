60 /**
61  * Run a job continuously, with input to/from the two specified files.
62  * The job should already be set up, and must be free by the caller
63  * after return.
64  *
65  * Buffers of ::rs_inbuflen and ::rs_outbuflen are allocated for
66  * temporary storage.
67  *
68  * \param in_file Source of input bytes, or NULL if the input buffer
69  * should not be filled.
70  *
71  * \return RS_DONE if the job completed, or otherwise an error result.
72  */
73 rs_result
74 rs_whole_run(rs_job_t *job, FILE *in_file, FILE *out_file)
75 {
76  rs_buffers_t buf;
77  rs_result result;
78  rs_filebuf_t *in_fb = NULL, *out_fb = NULL;
79 
80  if (in_file)
81  in_fb = rs_filebuf_new(in_file, rs_inbuflen);
82 
83  if (out_file)
84  out_fb = rs_filebuf_new(out_file, rs_outbuflen);
85 
86  result = rs_job_drive(job, &buf,
87  in_fb ? rs_infilebuf_fill : NULL, in_fb,
88  out_fb ? rs_outfilebuf_drain : NULL, out_fb);
89 
90  if (in_fb)
91  rs_filebuf_free(in_fb);
92 
93  if (out_fb)
94  rs_filebuf_free(out_fb);
95 
96  return result;
97 }
