23 void rs_hexify(char *to_buf, void const *from, int from_len)
24 {
25  static const char hex_chars[] = "0123456789abcdef";
26  unsigned char const *from_buf = (unsigned char const *)from;
27 
28  while (from_len-- > 0) {
29  *(to_buf++) = hex_chars[((*from_buf) >> 4) & 0xf];
30  *(to_buf++) = hex_chars[(*from_buf) & 0xf];
31  from_buf++;
32  }
33 
34  *to_buf = 0;
35 }
