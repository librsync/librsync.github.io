22void rs_hexify(char *to_buf, void const *from, int from_len)
23{
24 static const char hex_chars[] = "0123456789abcdef";
25 unsigned char const *from_buf = (unsigned char const *)from;
26
27 while (from_len-- > 0) {
28 *(to_buf++) = hex_chars[((*from_buf) >> 4) & 0xf];
29 *(to_buf++) = hex_chars[(*from_buf) & 0xf];
30 from_buf++;
31 }
32
33 *to_buf = 0;
34}
