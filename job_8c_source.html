110 static rs_result rs_job_complete(rs_job_t *job, rs_result result)
111 {
112  rs_job_check(job);
113 
114  job->statefn = rs_job_s_complete;
115  job->final_result = result;
116 
117  if (result != RS_DONE) {
118  rs_error("%s job failed: %s", job->job_name, rs_strerror(result));
119  } else {
120  rs_trace("%s job complete", job->job_name);
121  }
122 
123  job->stats.end = time(NULL);
124  if (result == RS_DONE && !rs_tube_is_idle(job))
125  /* Processing is finished, but there is still some data
126  * waiting to get into the output buffer. */
127  return RS_BLOCKED;
128  else
129  return result;
130 }

133 rs_result rs_job_iter(rs_job_t *job, rs_buffers_t *buffers)
134 {
135  rs_result result;
136  rs_long_t orig_in, orig_out;
137 
138  orig_in = buffers->avail_in;
139  orig_out = buffers->avail_out;
140 
141  result = rs_job_work(job, buffers);
142 
143  if (result == RS_BLOCKED || result == RS_DONE)
144  if ((orig_in == buffers->avail_in) && (orig_out == buffers->avail_out)
145  && orig_in && orig_out) {
146  rs_log(RS_LOG_ERR, "internal error: job made no progress "
147  "[orig_in=" PRINTF_FORMAT_U64 ", orig_out=" PRINTF_FORMAT_U64 ", final_in=" PRINTF_FORMAT_U64 ", final_out=" PRINTF_FORMAT_U64 "]",
148  PRINTF_CAST_U64(orig_in), PRINTF_CAST_U64(orig_out), PRINTF_CAST_U64(buffers->avail_in),
149  PRINTF_CAST_U64(buffers->avail_out));
150  return RS_INTERNAL_ERROR;
151  }
152 
153  return result;
154 }

157 static rs_result
158 rs_job_work(rs_job_t *job, rs_buffers_t *buffers)
159 {
160  rs_result result;
161 
162  rs_job_check(job);
163 
164  if (!buffers) {
165  rs_error("NULL buffer passed to rs_job_iter");
166  return RS_PARAM_ERROR;
167  }
168  job->stream = buffers;
169 
170  while (1) {
171  result = rs_tube_catchup(job);
172  if (result == RS_BLOCKED)
173  return result;
174  else if (result != RS_DONE)
175  return rs_job_complete(job, result);
176 
177  if (job->statefn == rs_job_s_complete) {
178  if (rs_tube_is_idle(job))
179  return RS_DONE;
180  else
181  return RS_BLOCKED;
182  } else {
183  result = job->statefn(job);
184  if (result == RS_RUNNING)
185  continue;
186  else if (result == RS_BLOCKED)
187  return result;
188  else
189  return rs_job_complete(job, result);
190  }
191  }
192 
193  /* TODO: Before returning, check that we actually made some
194  * progress. If not, and we're not returning an error, this is a
195  * bug. */
196 }

214 rs_result
215 rs_job_drive(rs_job_t *job, rs_buffers_t *buf,
216  rs_driven_cb in_cb, void *in_opaque,
217  rs_driven_cb out_cb, void *out_opaque)
218 {
219  rs_result result, iores;
220 
221  rs_bzero(buf, sizeof *buf);
222 
223  do {
224  if (!buf->eof_in && in_cb) {
225  iores = in_cb(job, buf, in_opaque);
226  if (iores != RS_DONE)
227  return iores;
228  }
229 
230  result = rs_job_iter(job, buf);
231  if (result != RS_DONE && result != RS_BLOCKED)
232  return result;
233 
234  if (out_cb) {
235  iores = (out_cb)(job, buf, out_opaque);
236  if (iores != RS_DONE)
237  return iores;
238  }
239  } while (result != RS_DONE);
240 
241  return result;
242 }
