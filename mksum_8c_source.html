45 /** State of trying to send the signature header. \private */
46 static rs_result rs_sig_s_header(rs_job_t *job)
47 {
48  rs_signature_t *sig = job->signature;
49  rs_result result;
50 
51  if ((result =
52  rs_signature_init(sig, job->sig_magic, job->sig_block_len,
53  job->sig_strong_len, 0)) != RS_DONE)
54  return result;
55  rs_squirt_n4(job, sig->magic);
56  rs_squirt_n4(job, sig->block_len);
57  rs_squirt_n4(job, sig->strong_sum_len);
58  rs_trace("sent header (magic %#x, block len = %d, strong sum len = %d)",
59  sig->magic, sig->block_len, sig->strong_sum_len);
60  job->stats.block_len = sig->block_len;
61 
62  job->statefn = rs_sig_s_generate;
63  return RS_RUNNING;
64 }

66 /** Generate the checksums for a block and write it out. Called when we
67  * already know we have enough data in memory at \p block. \private */
68 static rs_result rs_sig_do_block(rs_job_t *job, const void *block, size_t len)
69 {
70  rs_signature_t *sig = job->signature;
71  rs_weak_sum_t weak_sum;
72  rs_strong_sum_t strong_sum;
73 
74  weak_sum = rs_signature_calc_weak_sum(sig, block, len);
75  rs_signature_calc_strong_sum(sig, block, len, &strong_sum);
76  rs_squirt_n4(job, weak_sum);
77  rs_tube_write(job, strong_sum, sig->strong_sum_len);
78  if (rs_trace_enabled()) {
79  char strong_sum_hex[RS_MAX_STRONG_SUM_LENGTH * 2 + 1];
80  rs_hexify(strong_sum_hex, strong_sum, sig->strong_sum_len);
81  rs_trace("sent block: weak=" FMT_WEAKSUM ", strong=%s", weak_sum,
82  strong_sum_hex);
83  }
84  job->stats.sig_blocks++;
85  return RS_RUNNING;
86 }

88 /** State of reading a block and trying to generate its sum. \private */
89 static rs_result rs_sig_s_generate(rs_job_t *job)
90 {
91  rs_result result;
92  size_t len;
93  void *block;
94 
95  /* must get a whole block, otherwise try again */
96  len = job->signature->block_len;
97  result = rs_scoop_read(job, len, &block);
98  /* If we are near EOF, get whatever is left. */
99  if (result == RS_INPUT_ENDED)
100  result = rs_scoop_read_rest(job, &len, &block);
101  if (result == RS_INPUT_ENDED) {
102  return RS_DONE;
103  } else if (result != RS_DONE) {
104  rs_trace("generate stopped: %s", rs_strerror(result));
105  return result;
106  }
107  rs_trace("got " FMT_SIZE " byte block", len);
108  return rs_sig_do_block(job, block, len);
109 }
