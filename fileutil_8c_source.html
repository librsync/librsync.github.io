45 /**
46  * \brief Open a file, with special handling for `-' or unspecified
47  * parameters on input and output.
48  *
49  * \param fopen-style mode string.
50  */
51 FILE *
52 rs_file_open(char const *filename, char const *mode, int force)
53 {
54  FILE *f;
55  int is_write;
56 
57  is_write = mode[0] == 'w';
58 
59  if (!filename || !strcmp("-", filename)) {
60  if (is_write) {
61 #if _WIN32
62  _setmode(_fileno(stdout), _O_BINARY);
63 #endif
64  return stdout;
65  } else {
66 #if _WIN32
67  _setmode(_fileno(stdin), _O_BINARY);
68 #endif
69  return stdin;
70  }
71  }
72 
73  if (!force && is_write) {
74  if ((f = fopen(filename, "rb"))) {
75  // File exists
76  rs_error("File exists \"%s\", aborting!", filename);
77  fclose(f);
78  exit(RS_IO_ERROR);
79  }
80  }
81 
82  if (!(f = fopen(filename, mode))) {
83  rs_error("Error opening \"%s\" for %s: %s", filename,
84  is_write ? "write" : "read",
85  strerror(errno));
86  exit(RS_IO_ERROR);
87  }
88 
89  return f;
90 }

92 int rs_file_close(FILE * f)
93 {
94  if ((f == stdin) || (f == stdout)) return 0;
95  return fclose(f);
