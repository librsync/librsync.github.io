32/** A simple 32bit checksum that can be incrementally updated. */
33rs_weak_sum_t rs_calc_weak_sum(weaksum_kind_t kind, void const *buf, size_t len)
34{
35 if (kind == RS_ROLLSUM) {
36 Rollsum sum;
37 RollsumInit(&sum);
38 RollsumUpdate(&sum, buf, len);
39 return RollsumDigest(&sum);
40 } else {
41 rabinkarp_t sum;
42 rabinkarp_init(&sum);
43 rabinkarp_update(&sum, buf, len);
44 return rabinkarp_digest(&sum);
45 }
46}

48/** Calculate and store into SUM a strong checksum.
49 *
50 * In plain rsync, the checksum is perturbed by a seed value. This is used when
51 * retrying a failed transmission: we've discovered that the hashes collided at
52 * some point, so we're going to try again with different hashes to see if we
53 * can get it right. (Check tridge's thesis for details and to see if that's
54 * correct.)
55 *
56 * Since we can't retry I'm not sure if it's very useful for librsync, except
57 * perhaps as protection against hash collision attacks. */
58void rs_calc_strong_sum(strongsum_kind_t kind, void const *buf, size_t len,
59 rs_strong_sum_t *sum)
60{
61 if (kind == RS_MD4) {
62 rs_mdfour((unsigned char *)sum, buf, len);
63 } else {
64 blake2b_state ctx;
65 blake2b_init(&ctx, RS_MAX_STRONG_SUM_LENGTH);
66 blake2b_update(&ctx, (const uint8_t *)buf, len);
67 blake2b_final(&ctx, (uint8_t *)sum, RS_MAX_STRONG_SUM_LENGTH);
68 }
69}
