26 /** Simple hashtable.
27  *
28  * This is a minimal hashtable containing pointers to arbitrary
29  * entries with configurable hashtable size and support for custom
30  * hash() and cmp() methods. The cmp() method can either be a simple
31  * comparison between two keys, or can be against a special match
32  * object containing additional mutable state. This allows for things
33  * like deferred and cached evaluation of costly comparison data.
34  * The hash() function doesn't need to avoid clustering behaviour.
35  *
36  * It uses open addressing with quadratic probing for collisions. The
37  * MurmurHash3 finalization function is used on the hash() output to
38  * avoid clustering. There is no support for removing entries, only
39  * adding them. Multiple entries with the same key can be added, and
40  * you can use a fancy cmp() function to find particular entries by
41  * more than just their key. There is an iterator for iterating
42  * through all entries in the hashtable. There are optional
43  * hashtable_find() find/match/hashcmp/entrycmp stats counters that
44  * can be disabled by defining HASHTABLE_NSTATS.
45  *
46  * Example:
47  *
48  * typedef ... key_t;
49  * int key_hash(const key_t *e);
50  * int key_cmp(key_t *e, const key_t *o);
51  *
52  * typedef struct entry {
53  * key_t key; // Inherit from key_t.
54  * ...extra entry value data...
55  * } entry_t;
56  * void entry_init(entry_t *e, ...);
57  *
58  * hashtable_t *t;
59  * entry_t entries[300];
60  * key_t k;
61  * entry_t *e;
62  *
63  * t = hashtable_new(300, &key_hash, &key_cmp);
64  * entry_init(&entries[5], ...);
65  * hashtable_add(t, &entries[5]);
66  * k = ...;
67  * e = hashtable_find(t, &k);
68  *
69  * hashtable_iter i;
70  * for (e = hashtable_iter(&i, t); e != NULL; e = hashtable_next(&i))
71  * ...
72  *
73  * hashtable_free(t);
74  *
75  * The hash() and cmp() fuctions will typically take pointers to
76  * key/entry instances the same as the pointers stored in the
77  * hashtable. However it is also possible for them to take "match
78  * objects" that are a "subclass" of the entry type that contain
79  * additional state for complicated comparision operations.
80  *
81  * Example:
82  *
83  * typedef struct match {
84  * key_t key; // Inherit from key_t;
85  * ...extra match criteria and state data...
86  * } match_t;
87  * int match_cmp(match_t *m, const entry_t *e);
88  *
89  * ...
90  * match_t m;
91  *
92  * t = hashtable_new(300, &key_hash, &match_cmp);
93  * ...
94  * m = ...;
95  * e = hashtable_find(t, &m);
96  *
97  * The cmp() function is only called for finding hashtable entries
98  * and can mutate the match_t object for doing things like deferred
99  * and cached evaluation of expensive match data. It can also access
100  * the whole entry_t object to match against more than just the key. */
101 
102 /** The hash() function type.
103  *
104  * Args:
105  * *k - the key or match object to hash.
106  *
107  * Returns:
108  * An unsigned hash value.
109  */
110 typedef unsigned (*hash_f) (const void *k);
111 
112 /** The cmp() function type.
113  *
114  * Args:
115  * *k - the key or match object to try and match to.
116  * *o - the key or entry object to match against.
117  *
118  * Returns:
119  * -1, 0, or 1 if *e is less, equal, or more that *o. */
120 typedef int (*cmp_f) (void *k, const void *o);

122 /** The hashtable type. */
123 typedef struct _hashtable {
124  int size; /* Size of allocated hashtable. */
125  int count; /* Number of entries in hashtable. */
126  hash_f hash; /* Function for hashing entries. */
127  cmp_f cmp; /* Function for comparing entries. */
128 #ifndef HASHTABLE_NSTATS
129  /* The following are for accumulating hashtable_find() stats. */
130  long find_count; /* The count of finds tried. */
131  long match_count; /* The count of matches found. */
132  long hashcmp_count; /* The count of hash compares done. */
133  long entrycmp_count; /* The count of entry compares done. */
134 #endif
135  void **etable; /* Table of pointers to entries. */
136  unsigned ktable[]; /* Table of hash keys. */
137 } hashtable_t;

145 /** Allocate and initialize a hashtable instance.
146  *
147  * The provided size is used as an indication of the number of
148  * entries you wish to add, but the allocated size will probably be
149  * larger depending on the implementation to enable optimisations or
150  * avoid degraded performance. It may be possible to fill the table
151  * beyond the requested size, but performance can start to degrade
152  * badly if it is over filled.
153  *
154  * Args:
155  * size - The desired minimum size of the hash table.
156  * hash - The hash function to use.
157  * cmp - The compare function to use.
158  *
159  * Returns:
160  * The initialized hashtable instance or NULL if it failed. */
161 hashtable_t *hashtable_new(int size, hash_f hash, cmp_f cmp);

163 /** Destroy and free a hashtable instance.
164  *
165  * This will free the hashtable, but will not free the entries in the
166  * hashtable. If you want to free the entries too, use a hashtable
167  * iterator to free the the entries first.
168  *
169  * Args:
170  * *t - The hashtable to destroy and free. */

173 /** Add an entry to a hashtable.
174  *
175  * This doesn't use cmp() or do any checks for existing copies or
176  * instances, so it will add duplicates. If you want to avoid adding
177  * duplicates, use hashtable_find() to check for existing entries
178  * first.
179  *
180  * Args:
181  * *t - The hashtable to add to.
182  * *e - The entry object to add.
183  *
184  * Returns:
185  * The added entry, or NULL if the table is full. */
186 void *hashtable_add(hashtable_t *t, void *e);

188 /** Find an entry in a hashtable.
189  *

192  *
