26 /** \file hashtable.h
27  * A generic open addressing hashtable.
28  *
29  * This is a minimal hashtable containing pointers to arbitrary
30  * entries with configurable hashtable size and support for custom
31  * hash() and cmp() methods. The cmp() method can either be a simple
32  * comparison between two keys, or can be against a special match
33  * object containing additional mutable state. This allows for things
34  * like deferred and cached evaluation of costly comparison data. The
35  * hash() function doesn't need to avoid clustering behaviour.
36  *
37  * It uses open addressing with quadratic probing for collisions. The
38  * MurmurHash3 finalization function is used on the hash() output to
39  * avoid clustering. There is no support for removing entries, only
40  * adding them. Multiple entries with the same key can be added, and
41  * you can use a fancy cmp() function to find particular entries by
42  * more than just their key. There is an iterator for iterating
43  * through all entries in the hashtable. There are optional
44  * hashtable_find() find/match/hashcmp/entrycmp stats counters that
45  * can be disabled by defining HASHTABLE_NSTATS.
46  *
47  * The types and methods of the hashtable and its contents are
48  * specified by using \#define parameters set to their basenames (the
49  * prefixes for the *_t type and *_func() methods) before doing
50  * \#include "hashtable.h". This produces static inline type-safe
51  * methods that are either application optimized for speed or
52  * wrappers around void* implementation methods for compactness.
53  *
54  * \param ENTRY - the entry type basename.
55  * \param KEY - optional key type basename (default: ENTRY).
56  * \param MATCH - optional match type basename (default: KEY).
57  * \param NAME - optional hashtable type basename (default: ENTRY_hashtable).
58  *
59  * Example: \code
60  * typedef ... key_t;
61  * int key_hash(const key_t *e);
62  * int key_cmp(key_t *e, const key_t *o);
63  *
64  * typedef struct entry {
65  * key_t key; // Inherit from key_t.
66  * ...extra entry value data...
67  * } entry_t;
68  * void entry_init(entry_t *e, ...);
69  *
70  * #define ENTRY entry
71  * #define KEY key
72  * #include "hashtable.h"
73  *
74  * hashtable_t *t;
75  * entry_t entries[300];
76  * key_t k;
77  * entry_t *e;
78  *
79  * t = entry_hashtable_new(300);
80  * entry_init(&entries[5], ...);
81  * entry_hashtable_add(t, &entries[5]);
82  * k = ...;
83  * e = entry_hashtable_find(t, &k);
84  *
85  * hashtable_iter i;
86  * for (e = entry_hashtable_iter(&i, t); e != NULL; e = entry_hashtable_next(&i))
87  * ...
88  *
89  * entry_hashtable_free(t);
90  * \endcode
91  *
92  * The key_hash() and key_cmp() fuctions will typically take pointers
93  * to key/entry instances the same as the pointers stored in the
94  * hashtable. However it is also possible for them to take "match
95  * objects" that are a "subclass" of the entry type that contain
96  * additional state for complicated comparision operations.
97  *
98  * Example: \code
99  * typedef struct match {
100  * key_t key; // Inherit from key_t;
101  * ...extra match criteria and state data...
102  * } match_t;
103  * int match_cmp(match_t *m, const entry_t *e);
104  *
105  * #define ENTRY entry
106  * #define KEY key
107  * #define MATCH match
108  * #include "hashtable.h"
109  *
110  * ...
111  * match_t m;
112  *
113  * t = entry_hashtable_new(300);
114  * ...
115  * m = ...;
116  * e = entry_hashtable_find(t, &m);
117  * \endcode
118  *
119  * The match_cmp() function is only called for finding hashtable
120  * entries and can mutate the match_t object for doing things like
121  * deferred and cached evaluation of expensive match data. It can
122  * also access the whole entry_t object to match against more than
123  * just the key. */

152 /** MurmurHash3 finalization mix function. */
153 static inline unsigned mix32(unsigned int h)
154 {
155  h ^= h >> 16;
156  h *= 0x85ebca6b;
157  h ^= h >> 13;
158  h *= 0xc2b2ae35;
159  h ^= h >> 16;
160  return h;
161 }

192 /* Loop macro for probing table t for key k, setting hk to the hash for k
193  reserving zero for empty buckets, and iterating with index i and entry
194  hash h, terminating at an empty bucket. */
195 #define _for_probe(t, k, hk, i, h) \
196  const unsigned mask = t->size - 1;\
197  unsigned hk = KEY_HASH((KEY_T *)k), i, s, h;\
198  hk = hk ? hk : -1;\
199  for (i = mix32(hk) & mask, s = 0; (h = t->ktable[i]); i = (i + ++s) & mask)

237 /** Add an entry to a hashtable.
238  *
239  * This doesn't use MATCH_CMP() or do any checks for existing copies or
240  * instances, so it will add duplicates. If you want to avoid adding
241  * duplicates, use hashtable_find() to check for existing entries
242  * first.
243  *
244  * \param *t - The hashtable to add to.
245  * \param *e - The entry object to add.
246  *
247  * \return The added entry, or NULL if the table is full. */
248 static inline ENTRY_T *_FUNC(_add)(hashtable_t *t, ENTRY_T *e)
249 {
250  assert(e != NULL);
251  if (t->count + 1 == t->size)
252  return NULL;
253  _for_probe(t, e, he, i, h);
254  t->count++;
255  t->ktable[i] = he;
256  return t->etable[i] = e;
257 }

259 /** Find an entry in a hashtable.
260  *
261  * Uses MATCH_CMP() to find the first matching entry in the table in the
262  * same hash() bucket.
263  *
264  * \param *t - The hashtable to search.
265  * \param *m - The key or match object to search for.
266  *
267  * \return The first found entry, or NULL if nothing was found. */
268 static inline ENTRY_T *_FUNC(_find)(hashtable_t *t, MATCH_T *m)
269 {
270  assert(m != NULL);
271  ENTRY_T *e;
272 
273  _stats_inc(t->find_count);
274  _for_probe(t, m, hm, i, he) {
275  _stats_inc(t->hashcmp_count);
276  if (hm == he) {
277  _stats_inc(t->entrycmp_count);
278  if (!MATCH_CMP(m, e = t->etable[i])) {
279  _stats_inc(t->match_count);
280  return e;
281  }
282  }
283  }
284  return NULL;
285 }
